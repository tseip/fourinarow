69 bitset last_col;
70 for (size_t i = 0; i < HEIGHT; ++i) {
71 last_col.set((i + 1U) * WIDTH - 1U);

187 * Check for a vertical win. Drag the pattern against itself vertically

189 * starting locations are flagged.

192 static const bitset vertical_mask = []() {
193 bitset vertical_mask;
194 for (std::size_t row = 0; row < HEIGHT - (N - 1U); ++row) {
