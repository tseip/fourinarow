303 const std::size_t param_pack_idx = i;
304 for (std::size_t j = 0; j < num_param_packs; ++j) {
305 add_feature_group(params[param_pack_idx + j],
306 params[param_pack_idx + j + num_param_packs],
307 params[param_pack_idx + j + 2 * num_param_packs]);

310 lapse = std::bernoulli_distribution(lapse_rate);

312 vtile[i] = 1.0 / sqrt(pow(i / Board::get_board_width() - 1.5, 2) +
313 pow(i % Board::get_board_width() - 4.0, 2));
314 c_self = 2.0 * opp_scale / (1.0 + opp_scale);
315 c_opp = 2.0 / (1.0 + opp_scale);

380 double evaluate(const Board& b) const {
381 const Player player = b.active_player();
382 const Player other_player = get_other_player(player);
383 double val = 0.0;
384
385 for (const auto i : b.get_pieces(player).get_all_position_indices()) {
386 val += center_weight * vtile[i];
387 }
388
389 for (const auto i : b.get_pieces(other_player).get_all_position_indices()) {
390 val -= center_weight * vtile[i];
391 }
392
393 const auto player_pieces = feature_evaluator.query_pieces(b, player);
394 const auto opponent_pieces =
395 feature_evaluator.query_pieces(b, other_player);
396 const auto spaces = feature_evaluator.query_spaces(b);
397 for (const auto& feature : features) {
398 if (!feature.enabled) continue;
399 const auto i = feature.vector_index;
400 if (feature.feature.contained_in(player_pieces[i], spaces[i])) {
401 val += feature_group_weights[feature.weight_index].weight_act;
402 } else if (feature.feature.contained_in(opponent_pieces[i], spaces[i])) {
403 val -= feature_group_weights[feature.weight_index].weight_pass;
404 }
405 }
406 return player == Player::Player1 ? val : -val;
407 }

422 std::vector<typename Board::MoveT> get_moves(const Board& b,
423 Player evalPlayer,
424 bool sorted = true) {
425 const Player player = b.active_player();
426 const Player other_player = get_other_player(player);
427 const double c_act = (player == evalPlayer) ? c_self : c_opp;
428 const double c_pass = (player == evalPlayer) ? c_opp : c_self;
429
430 auto player_pieces = feature_evaluator.query_pieces(b, player);
431 auto opponent_pieces = feature_evaluator.query_pieces(b, other_player);
432 auto spaces = feature_evaluator.query_spaces(b);
433
434 std::unordered_map<typename Board::PatternT, typename Board::MoveT,
435 typename Board::PatternHasherT>
436 candidate_moves;
437 double deltaL = 0.0;
438 for (const auto& feature : features) {
439 if (!feature.enabled) continue;
440 const auto i = feature.vector_index;
441 if (feature.feature.contained_in(player_pieces[i], spaces[i])) {
442 deltaL -= c_pass *
443 feature_group_weights[feature.weight_index].diff_act_pass();
444 } else if (feature.feature.contained_in(opponent_pieces[i], spaces[i])) {
445 deltaL -=
446 c_act * feature_group_weights[feature.weight_index].diff_act_pass();
447 }
448 }
449
450 for (const auto i : b.get_spaces().get_all_position_indices()) {
451 candidate_moves[typename Board::PatternT(1LLU << i)] =
452 typename Board::MoveT(i,
453 deltaL + center_weight * vtile[i] +
454 (noise_enabled ? noise(engine) : 0.0),
455 player);
456 }
457
458 for (const auto& feature : features) {
459 if (!feature.enabled) continue;
460 const auto i = feature.vector_index;
461
462 // If either player can fill in the feature, and the current player
463 // can complete it...
464 if (feature.feature.can_be_completed(player_pieces[i], opponent_pieces[i],
465 spaces[i])) {
466 const typename Board::PatternT player_missing_pieces =
467 feature.feature.missing_pieces(b, player);
468 auto search = candidate_moves.find(player_missing_pieces);
469 if (search != candidate_moves.end()) {
470 search->second.val +=
471 c_pass * feature_group_weights[feature.weight_index].weight_pass;
472 }
473 }
474
475 // If the current player has the required pieces but the opponent can
476 // block us or if the other player has the feature and we can block
477 // them...
478 const bool can_be_removed =
479 feature.feature.can_be_removed(player_pieces[i], spaces[i]);
480 const bool can_remove_opponent =
481 feature.feature.can_be_removed(opponent_pieces[i], spaces[i]);
482 if (can_be_removed || can_remove_opponent) {
483 for (const auto& position :
484 feature.feature.spaces.get_all_positions()) {
485 if (b.contains_spaces(position)) {
486 auto search = candidate_moves.find(position);
487 if (search != candidate_moves.end()) {
488 if (can_be_removed)
489 search->second.val -=
490 c_pass *
491 feature_group_weights[feature.weight_index].weight_pass;
492 if (can_remove_opponent)
493 search->second.val +=
494 c_act *
495 feature_group_weights[feature.weight_index].weight_act;
496 }
497 }
498 }
499 }
500 }
501
502 std::vector<typename Board::MoveT> output_moves;
503 for (const auto kv : candidate_moves) {
504 output_moves.push_back(kv.second);
505 }
506 std::sort(output_moves.begin(), output_moves.end(),
507 [](const auto& m1, const auto& m2) {
508 return m1.board_position < m2.board_position;
509 });
510
511 if (!sorted) return output_moves;
512 std::sort(output_moves.begin(), output_moves.end(), std::greater<>());
513 return output_moves;
514 }

526 std::vector<typename Board::MoveT> get_pruned_moves(const Board& b,
527 Player evalPlayer) {
528 std::vector<typename Board::MoveT> candidates = get_moves(b, evalPlayer);
529 std::size_t i = 1;
530 while (i < candidates.size() &&
531 abs(candidates[0].val - candidates[i].val) < pruning_thresh) {
532 ++i;
533 }
534 if (i < candidates.size())
535 candidates.erase(candidates.begin() + i, candidates.end());
536 return candidates;
537 }

545 typename Board::MoveT get_random_move(const Board& b) {
